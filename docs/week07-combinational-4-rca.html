  • 完成功能模擬測試

非同步電路設計的基礎元件 C-Element (CE) - ce1.v

`timescale 1ns / 1ps

module ce1(
    input a,
    input b,
    output x
    );
    
    reg x;
    
    always@(a or b) begin
        if (a==1 && b==1)
            x = 1'b1;
        else if (a==0 && b==0)
            x = 1'b0; 
    end
    
endmodule

C-Element (CE) RTL分析電路 (Schematic)

測試程式 ce_tb.v 

`timescale 1ns / 1ps

module ce1_tb;

    reg a,b;
    wire x;

    ce1 uut1 (.a(a),.b(b),.x(x));
    
    initial begin
        {a,b} = 2'b00;
        #10;     
        {a,b} = 2'b10; #10;
        {a,b} = 2'b01; #10;
        {a,b} = 2'b11; #10;
        {a,b} = 2'b10; #10;
        {a,b} = 2'b01; #10;
        {a,b} = 2'b00; #10;
        #100
        $finish;
    end

endmodule
